//Program to perform selection sorting 
#include<stdio.h>
int main()
{
	int a[50],i,j,n;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			if(a[j]<a[i])
			{
				temp=a[i];	
				a[i]=a[j];
				a[j]=temp;
			}
		}
	}
	printf("Sorted array :- ");
	for(i=0;i<n;i++)
	printf("%d ",a[i]);
	getch();
	return 0;
}
